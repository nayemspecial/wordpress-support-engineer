```
function evenOdd(inputString) {
  const length = inputString.length;
  if (length % 2 === 0) {
    return 'even';
  } else {
    return 'odd';
  }
}

// Example usage:
const input1 = 'JavaScript';
const input2 = 'Hello';

console.log(evenOdd(input1)); // Output: 'even'
console.log(evenOdd(input2)); // Output: 'odd'
```
